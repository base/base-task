## Usage

```js
var Base = require('base-methods');
var tasks = require('{%= name %}');
Base.use(tasks());

var base = new Base();
base.task('foo', function(cb) {
  console.log('this is foo!');
  cb();
});
base.task('bar', function(cb) {
  console.log('this is bar!');
  cb();
});

base.build(['foo', 'bar'], function(err) {
  // this is foo!
  // this is bar!
  console.log('done!');
});
```
